interface INonfungiblePositionManager {
    struct MintParams {
        address token0; address token1; uint24 fee;
        int24 tickLower; int24 tickUpper;
        uint256 amount0Desired; uint256 amount1Desired;
        uint256 amount0Min; uint256 amount1Min;
        address recipient; uint256 deadline;
    }
    function createAndInitializePoolIfNecessary(address token0, address token1, uint24 fee, uint160 sqrtPriceX96) external returns (address pool);
    function mint(MintParams calldata params) external returns (uint256 tokenId, uint128 liquidity, uint256 amount0, uint256 amount1);
}

// --- КОНТРАКТ ТОКЕНА С ЗАЩИТОЙ ---
contract MemeToken is ERC20 {
    address public factory;
    bool public tradingStarted = false;

    constructor(string memory name, string memory symbol, uint256 totalSupply, address _factory) ERC20(name, symbol) {
        factory = _factory;
        _mint(_factory, totalSupply);
    }

    function startTrading() external {
        require(msg.sender == factory, "Only factory");
        tradingStarted = true;
    }

    // Исправлено для OpenZeppelin 5.0+: используем _update вместо _beforeTokenTransfer
    function _update(address from, address to, uint256 value) internal virtual override {
        if (!tradingStarted && from != address(0)) {
            require(from == factory || to == factory, "Locked until migration");
        }
        super._update(from, to, value);
    }
}

// --- ОСНОВНАЯ ФАБРИКА ---
contract GemFunFactoryFinal is Ownable, ReentrancyGuard, ERC1155Holder {
    using SafeERC20 for IERC20;

    address public constant HASH = 0x81814Cd2EC975eAc745578066963DE4f8A7De293;
    address public constant INVENTORY = 0x13CE10a3e09FA8000BA8A13fCbe8813f476584e7; 
    address public constant V3_POSITION_MANAGER = 0x03a520b32C04BF3bEEf7BEb72E919cf822Ed34f1;
    address public constant DEAD_ADDRESS = 0x000000000000000000000000000000000000dEaD;

    uint256 public deployFee = 0.0001 ether; 
    uint256 public constant TOTAL_SUPPLY = 1_000_000_000 ether;      
    uint256 public constant CURVE_SUPPLY = 300_000_000 ether;      
    uint256 public constant LIQUIDITY_RESERVE = 300_000_000 ether; 
    uint256 public constant MINING_RESERVE_INIT = 400_000_000 ether; 

    uint256[] public rates = [0.07 ether, 0.7 ether, 4.033 ether, 0.558 ether, 8.416 ether, 40.333 ether];

    struct StakerInfo {
        uint256 totalHashrate; uint256 lastUpdate; uint256 rewardDebt;
        uint256[6] stakedAmounts;
    }

    struct TokenInfo {
        bool migrated; bool curveCompleted; uint256 sold; uint256 raised; uint256 miningReserve;
        mapping(address => StakerInfo) stakers;
    }

    struct TokenCore { string name; string symbol; bytes32 logoHash; string description; }
    struct TokenMetadata { string website; string twitter; string telegram; string guild; }

    mapping(address => TokenInfo) public tokens;
    mapping(address => TokenCore) public tokenCore;
    mapping(address => TokenMetadata) public tokenMetadata; 
    mapping(address => address) public tokenCreator;
    mapping(address => uint256) public userTokensCount;
    mapping(address => uint256) public userMigratedCount; 
    address[] public allTokens;

    event TokenCreated(address indexed token, address indexed creator);
    event Trade(address indexed token, address indexed user, bool isBuy, uint256 hashAmt, uint256 memeAmt);
    event Migrated(address indexed token, uint256 lpTokenId);
    event TokenRemoved(address indexed token);

    constructor() Ownable(msg.sender) {}

    modifier onlyValidToken(address tokenAddr) {
        require(tokenCreator[tokenAddr] != address(0), "Removed");
        _ ;
    }

    // --- УПРАВЛЕНИЕ ---
    function setDeployFee(uint256 newFee) external onlyOwner {
        deployFee = newFee;
    }

    function removeToken(address tokenAddr) external onlyOwner {
        require(tokenCreator[tokenAddr] != address(0), "Not exists");
        tokenCreator[tokenAddr] = address(0); 
        uint256 len = allTokens.length;
        for (uint256 i = 0; i < len; i++) {
            if (allTokens[i] == tokenAddr) {
                allTokens[i] = allTokens[len - 1];
                allTokens.pop();
                break;
            }
        }
        emit TokenRemoved(tokenAddr);
    }

    // --- МАТЕМАТИКА ---
    function _sqrt(uint256 y) internal pure returns (uint256 z) {
        if (y > 3) {
            z = y; uint256 x = y / 2 + 1;
            while (x < z) { z = x; x = (y / x + x) / 2; }
        } else if (y != 0) { z = 1; }
    }

    function encodePriceSqrt(uint256 amount1, uint256 amount0) internal pure returns (uint160) {
        return uint160(_sqrt((amount1 << 96) / amount0) << 48);
    }

    function calculateReserveFor(uint256 sold) public pure returns (uint256) {
        return (1e15 * sold / 1e18) + (sold * sold / (45000000000 * 1e18));
    }

    // --- ТОРГОВЛЯ ---
    function buy(address tokenAddr, uint256 memeOut, uint256 maxHashIn) external nonReentrant onlyValidToken(tokenAddr) {
        TokenInfo storage t = tokens[tokenAddr];
        require(!t.migrated && !t.curveCompleted, "Closed");
        uint256 actualHash = calculateReserveFor(t.sold + memeOut) - calculateReserveFor(t.sold);
        require(actualHash <= maxHashIn, "Slippage");
        t.sold += memeOut; t.raised += actualHash;
        if (t.sold >= CURVE_SUPPLY) t.curveCompleted = true;
        IERC20(HASH).safeTransferFrom(msg.sender, address(this), actualHash);
        IERC20(tokenAddr).safeTransfer(msg.sender, memeOut);
        emit Trade(tokenAddr, msg.sender, true, actualHash, memeOut);
    }

    function sell(address tokenAddr, uint256 memeIn, uint256 minHashOut) external nonReentrant onlyValidToken(tokenAddr) {
        TokenInfo storage t = tokens[tokenAddr];
        require(!t.migrated && !t.curveCompleted, "Closed");
        uint256 hashOut = calculateReserveFor(t.sold) - calculateReserveFor(t.sold - memeIn);
        require(hashOut >= minHashOut, "Slippage");
        t.sold -= memeIn; t.raised -= hashOut;
        IERC20(tokenAddr).safeTransferFrom(msg.sender, address(this), memeIn);
        IERC20(HASH).safeTransfer(msg.sender, hashOut);
        emit Trade(tokenAddr, msg.sender, false, hashOut, memeIn);
    }

    // --- МИГРАЦИЯ ---
    function migrate(address tokenAddr) external nonReentrant onlyValidToken(tokenAddr) {
        TokenInfo storage t = tokens[tokenAddr];
        require(t.curveCompleted && !t.migrated, "Not ready");
        t.migrated = true;

        (address t0, address t1) = HASH < tokenAddr ? (HASH, tokenAddr) : (tokenAddr, HASH);
        uint256 a0 = (t0 == HASH) ? t.raised : LIQUIDITY_RESERVE;
        uint256 a1 = (t1 == HASH) ? t.raised : LIQUIDITY_RESERVE;

        INonfungiblePositionManager manager = INonfungiblePositionManager(V3_POSITION_MANAGER);
        manager.createAndInitializePoolIfNecessary(t0, t1, 3000, encodePriceSqrt(a1, a0));
        
        IERC20(t0).forceApprove(V3_POSITION_MANAGER, a0);
        IERC20(t1).forceApprove(V3_POSITION_MANAGER, a1);

        manager.mint(INonfungiblePositionManager.MintParams({
            token0: t0, token1: t1, fee: 3000, tickLower: -887220, tickUpper: 887220,
            amount0Desired: a0, amount1Desired: a1, amount0Min: 0, amount1Min: 0,
            recipient: DEAD_ADDRESS, deadline: block.timestamp + 600
        }));

        // ОТКРЫВАЕМ ТОРГИ В САМЫЙ КОНЕЦ
        MemeToken(tokenAddr).startTrading();

        userMigratedCount[tokenCreator[tokenAddr]]++;
        emit Migrated(tokenAddr, 0);
    }

    // --- МАЙНИНГ ---
    function _updateRewards(address tokenAddr, address user) internal {
        TokenInfo storage t = tokens[tokenAddr];
        StakerInfo storage s = t.stakers[user];
        if (s.lastUpdate > 0 && s.totalHashrate > 0) {
            s.rewardDebt += ((block.timestamp - s.lastUpdate) * s.totalHashrate) / 60;
        }
        s.lastUpdate = block.timestamp;
    }

    function stake(address tokenAddr, uint256 id, uint256 amount) external nonReentrant onlyValidToken(tokenAddr) {
        require(tokens[tokenAddr].migrated, "Not migrated");
        require(id < 6, "ID");
        _updateRewards(tokenAddr, msg.sender);
        StakerInfo storage s = tokens[tokenAddr].stakers[msg.sender];
        s.stakedAmounts[id] += amount;
        s.totalHashrate += amount * rates[id];
        IERC1155(INVENTORY).safeTransferFrom(msg.sender, address(this), id, amount, "");
    }

    function withdraw(address tokenAddr, uint256 id, uint256 amount) external nonReentrant {
        StakerInfo storage s = tokens[tokenAddr].stakers[msg.sender];
        require(s.stakedAmounts[id] >= amount, "Insuff.");
        _updateRewards(tokenAddr, msg.sender);
        s.stakedAmounts[id] -= amount;
        s.totalHashrate -= amount * rates[id];
        IERC1155(INVENTORY).safeTransferFrom(address(this), msg.sender, id, amount, "");
    }

    function claim(address tokenAddr) external nonReentrant {
        TokenInfo storage t = tokens[tokenAddr];
        _updateRewards(tokenAddr, msg.sender);
        uint256 reward = t.stakers[msg.sender].rewardDebt;
        require(reward > 0 && t.miningReserve >= reward, "Empty");
        t.stakers[msg.sender].rewardDebt = 0;
        t.miningReserve -= reward;
        IERC20(tokenAddr).safeTransfer(msg.sender, reward);
    }

    // --- ДЕПЛОЙ ---
    function deployToken(string calldata name, string calldata symbol, bytes32 logoHash, string calldata desc, uint256 creatorMemeBuy) external payable nonReentrant returns (address) {
        require(msg.value >= deployFee, "Fee");
        (bool success, ) = owner().call{value: msg.value}("");
        require(success, "Transfer failed");

        MemeToken token = new MemeToken(name, symbol, TOTAL_SUPPLY, address(this));
        address tokenAddr = address(token);
        
        TokenInfo storage t = tokens[tokenAddr];
        t.miningReserve = MINING_RESERVE_INIT;
        tokenCreator[tokenAddr] = msg.sender;
        tokenCore[tokenAddr] = TokenCore(name, symbol, logoHash, desc);
        allTokens.push(tokenAddr);
        userTokensCount[msg.sender]++;

        if (creatorMemeBuy > 0) {
            require(creatorMemeBuy <= CURVE_SUPPLY * 5 / 100, "5% Max");
            uint256 cost = calculateReserveFor(creatorMemeBuy);
            t.sold = creatorMemeBuy;
            t.raised = cost;
            IERC20(HASH).safeTransferFrom(msg.sender, address(this), cost);
            IERC20(tokenAddr).safeTransfer(msg.sender, creatorMemeBuy);
            emit Trade(tokenAddr, msg.sender, true, cost, creatorMemeBuy);
        }

        emit TokenCreated(tokenAddr, msg.sender);
        return tokenAddr;
    }

    function setTokenMetadata(address token, TokenMetadata calldata meta) external onlyValidToken(token) {
        require(msg.sender == tokenCreator[token], "Not creator");
        tokenMetadata[token] = meta;
    }

    // --- ВЬЮ-ФУНКЦИИ ---
    function getTokenFullData(address token) external view returns (TokenCore memory core, TokenMetadata memory meta, address creator, uint256[5] memory stats) {
        TokenInfo storage info = tokens[token];
        stats[0] = info.migrated ? 1 : 0;
        stats[1] = info.curveCompleted ? 1 : 0;
        stats[2] = info.sold; stats[3] = info.raised; stats[4] = info.miningReserve;
        return (tokenCore[token], tokenMetadata[token], tokenCreator[token], stats);
    }

    function getAccountData(address tokenAddr, address user) external view returns (uint256 walletBalance, uint256 totalHashrate, uint256 pendingRewards, uint256[6] memory stakedItems) {
        walletBalance = IERC20(tokenAddr).balanceOf(user);
        TokenInfo storage t = tokens[tokenAddr];
        StakerInfo storage s = t.stakers[user];
        totalHashrate = s.totalHashrate;
        uint256 rewards = s.rewardDebt;
        if (s.lastUpdate > 0 && s.totalHashrate > 0) {
            rewards += ((block.timestamp - s.lastUpdate) * s.totalHashrate) / 60;
        }
        pendingRewards = rewards;
        stakedItems = s.stakedAmounts;
    }

    function getTokensPage(uint256 offset, uint256 limit) external view returns (address[] memory) {
        uint256 total = allTokens.length;
        if (offset >= total) return new address[](0);
        uint256 size = (offset + limit > total) ? total - offset : limit;
        address[] memory page = new address[](size);
        for (uint256 i = 0; i < size; i++) {
            page[i] = allTokens[offset + i];
        }
        return page;
    }
}
